COMMANDS:

addducks              # add or subtract ducks
setducks              # set amount of ducks value
setdpc                # set duck per click
adddpc                # add or subtract duck per click
setdpcmulitiplier     # set duck per click multiplier
setdpsmultiplier      # set duck per second mulitplier
setdps                # set duck per second
adddps                # add or subtract duck per second
simulate              # simulate x amount of time, so simulate 60 simulates 60 seconds of gameplay instantly. By seconds.
globalgamespeed       # makes global game speed to whatever number, 1 is default speed, which is 60 ticks.
close                 # close console
quit                  # quit game
save                  # save game
reset                 # reset all game values
stats                 # display current stats 
help                  # display basic commands
